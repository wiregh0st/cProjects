CREATE TABLE User_account (
id INT UNSIGNED NOT NULL AUTO_INCREMENT,
email VARCHAR(255) NOT NULL UNIQUE,
firstName VARCHAR(50),
lastName VARCHAR(50),
phone VARCHAR(10),
streetAddress VARCHAR(85),
city VARCHAR(85),
state CHAR(2),
zip CHAR(5),
salt CHAR(22),
hash CHAR(32),
PRIMARY KEY (id)
);

CREATE TABLE Subscription_type (
id INT UNSIGNED NOT NULL AUTO_INCREMENT,
name VARCHAR(50) NOT NULL,
screens INT,
price CHAR(10),
description VARCHAR(225),
PRIMARY KEY (id)
); 

CREATE TABLE Payment_method (
id INT UNSIGNED NOT NULL AUTO_INCREMENT,
creditCardNumber CHAR(19) UNIQUE,
cvv VARCHAR(4),
expirationDate CHAR(5),
acctID INT UNSIGNED NOT NULL UNIQUE,
PRIMARY KEY (id),
FOREIGN KEY(acctID) REFERENCES User_account(id)
  ON DELETE RESTRICT ON UPDATE RESTRICT
);

CREATE TABLE Invoice (
id INT UNSIGNED NOT NULL AUTO_INCREMENT,
invoiceID VARCHAR(22) NOT NULL UNIQUE,
serviceStart DATE,
acctID INT UNSIGNED NOT NULL,
subID INT UNSIGNED NOT NULL,
paymentID INT UNSIGNED NOT NULL,
purchaseAmount DECIMAL(6,3),
PRIMARY KEY (id),
FOREIGN KEY (acctID) REFERENCES User_account(id)
  ON DELETE RESTRICT ON UPDATE RESTRICT,
FOREIGN KEY (subID) REFERENCES Subscription_type(id)
  ON DELETE RESTRICT ON UPDATE RESTRICT,
FOREIGN KEY (paymentID) REFERENCES Payment_method(id)
  ON DELETE RESTRICT ON UPDATE RESTRICT
);
